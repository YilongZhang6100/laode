\begin{exercise} \label{c4.9.1}
Find the inverse of the matrix
\[
\mattwo{2}{1}{3}{2}.
\]

\begin{solution}
\soln
Use \eqref{e:formAinv} to compute the inverse of the matrix,
as follows:
\[ \mattwo{2}{1}{3}{2}^{-1} = \frac{1}{4-3}\mattwo{2}{-1}{-3}{2}
= \mattwo{2}{-1}{-3}{2}. \]

\end{solution}
\end{exercise}
