\begin{exercise} \label{c4.8.1}
Verify by matrix multiplication that the following matrices are inverses of each other:
\[
\left( \begin{array}{rrr}
1  &  0  &  2\\
0  & -1  &  2\\
1  &  0  &  1
\end{array} \right)\AND
\left( \begin{array}{rrr}
-1 &   0 &   2\\
 2 &  -1 &  -2\\
 1 &   0 &  -1
\end{array} \right).
\]

\begin{solution}
\soln
If two matrices are inverses of each other, then their product is the
identity matrix.  So:
\[ \matthree{1}{0}{2}{0}{-1}{2}{1}{0}{1}
\matthree{-1}{0}{2}{2}{-1}{-2}{1}{0}{-1} =
\matthree{1}{0}{0}{0}{1}{0}{0}{0}{1}. \]

\end{solution}
\end{exercise}
