\begin{exercise} \label{c4.2.2a}
Find a $2\times 2$ matrix that reflects vectors in the $(x,y)$ plane across
the $x$ axis.

\begin{solution}
\soln The map $L_A$ that reflects vectors across the $x$-axis is
$(x,y) \rightarrow (x,-y)$.  The matrix is
\[
A = \mattwo{1}{0}{0}{-1}.
\]

\end{solution}
\end{exercise}
