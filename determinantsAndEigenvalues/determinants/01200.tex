\begin{exercise} \label{YZ_7.1_Vandermonde_a} 
Compute $\det(V_2)$ where
\[
V_2=\Matrix{x_1&1\\ x_2&1}. 
\]

\begin{solution}
\ans
\[
\det(V_2)=x_1-x_2.
\]

\soln
Calculate $\det(V_2)$ using the definition of determinant for a $2\times 2$ matrix.
\end{solution}
\end{exercise}
