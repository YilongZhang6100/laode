\begin{exercise} \label{c5.2.4}
Write the matrix $A=\mattwo{2}{2}{-3}{0}$ as a linear combination
of the matrices
\[
B=\mattwo{1}{1}{0}{0} \AND C=\mattwo{0}{0}{1}{0}.
\]

\begin{solution}
\ans
\[ A = \mattwo{2}{2}{-3}{0} = 2\mattwo{1}{1}{0}{0} -
3\mattwo{0}{0}{1}{0} = 2B - 3C. \]

\end{solution}
\end{exercise}
