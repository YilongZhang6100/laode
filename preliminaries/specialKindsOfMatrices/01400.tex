\begin{exercise} \label{YZ_1.3_skewsym2}
If $A$ is an $n\times n$ matrix, show that $A-A^t$ is skew-symmetric.

\begin{solution}
\soln According to \eqref{at+bt}, $(A-A^t)^t=A^t-(A^t)^t$. Since $(A^t)^t=A$, we have 
$(A-A^t)^t=A^t-A=-(A-A^t)$.
So $A-A^t$ is skew-symmetric.
\end{solution}
\end{exercise}
